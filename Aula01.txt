***criando tabela clientes***
CREATE TABLE CLIENTES (
DOCUMENTO INT PRIMARY KEY NOT NULL,
NOME VARCHAR (120) NOT NULL,
NOME_FANTASIA VARCHAR (120),
DATA_DE_NASCIMENTO DATE NOT NULL,
E_MAIL VARCHAR (255) NOT NULL,
TELEFONE VARCHAR (11) NOT NULL,
TIPO_DE_PESSOA CHAR (1) NOT NULL,
ATIVO INT NOT NULL)

***criando tabela conta_bancária***
CREATE TABLE CONTA_BANCÁRIA (
ID_CONTA_CLIENTE INT PRIMARY KEY NOT NULL,
DOCUMENTO_CLIENTE VARCHAR (120) FOREIGN KEY,
CÓDIGO_BANCO INT,
AGÊNCIA INT NOT NULL,
NÚMERO_CONTA INT NOT NULL,
TIPO_CONTA VARCHAR (8) NOT NULL,
ATIVO INT NOT NULL)
